#Check for Session and Workflow descriptions

SELECT 
REP_SUBJECT.SUBJECT_AREA AS FOLDER_NAME,
REP_ALL_TASKS.TASK_NAME, 
REP_ALL_TASKS.DESCRIPTION
FROM 
REP_SUBJECT REP_SUBJECT, 
REP_ALL_TASKS REP_ALL_TASKS
WHERE 
REP_SUBJECT.SUBJECT_ID = REP_ALL_TASKS.SUBJECT_ID AND 
REP_ALL_TASKS.TASK_TYPE IN (68,71) AND
REP_SUBJECT.SUBJECT_AREA = '{Folder_Name}' AND 
REP_ALL_TASKS.DESCRIPTION IS  NULL
ORDER BY FOLDER_NAME



#In place of {Folder_Name} you can mention the folder name where you want to check for the results.
